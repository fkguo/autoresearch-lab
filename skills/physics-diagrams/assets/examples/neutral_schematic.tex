\newcommand{\sourceDecay}[2]{%
  \begin{tikzpicture}[baseline=(c.base), line cap=round, line join=round, >=Stealth]
    \coordinate (c) at (0,0);
    \node[pdBlock, circle, fill=pdSource, minimum size=8.2mm] (src) at (0,0) {$#1$};
    \draw[pdSourceLine] (-1.25,0) -- (src.west);
    \draw[pdParticle] (src.east) ++(0,0.13) to[out=18,in=180] (1.32,0.45);
    \draw[pdParticle] (src.east) ++(0,-0.13) to[out=-18,in=180] (1.32,-0.45);
    \draw[pdExchange] (src.east) -- (1.32,0);
    \node[pdLabel, above=1.1mm] at (0.82,0.47) {$h_1$};
    \node[pdLabel, below=1.1mm] at (0.82,-0.47) {$h_2$};
    \if\relax\detokenize{#2}\relax\else
      \node[pdLabel, right=0.5mm] at (1.32,0) {$#2$};
    \fi
  \end{tikzpicture}%
}
